// Example of Airthmetic operators 

#include<stdio.h>
void main()
{
    int a,b;
    
    a = 10;
    b = 20;
    
    printf("%d + %d = %d",a,b,a+b);
    printf("\n%d - %d = %d",a,b,a-b);
    printf("\n%d * %d = %d",a,b,a*b);
    printf("\n%d / %d = %d",a,b,a/b);
    printf("\n%d %% %d = %d",a,b,a%b);
}

3) Relational operator : or (comparision operator)

    <   >    <=   >=   ==  != 

// Example of Relational operator 

#include<stdio.h>
void main()
{
    int num1,num2; // step 1 : variable declaration 
    
    num1 = 20;    // step 2 : variable defination 
    num2 = 34;
    
    // relational or comparision operator 
    // 0 - false  1 means true 
    
    int result;
    
    result = num1 > num2;
    printf("\n%d",result);
    
    result = num1 < num2;
    printf("\n%d",result);
    
    result = num1 == num2;
    printf("\n%d",result);
    
    result = num1 != num2;
    printf("\n%d",result);
}



4) logical operator : 

        &&   (And operator)

        ||   (OR operator)

        !    (not operator)
// Example of Logical operator 

#include<stdio.h>
void main()
{
    int num1,num2,num3; // step 1 : variable declaration 
    
    num1 = 20;    // step 2 : variable defination 
    num2 = 34;
    num3 = 40;
    
    int result = (num1>num2) && (num1 > num3);
    printf("\n%d",result);
    
    result = (num1<num2) && (num1 < num3);
    printf("\n%d",result);
    
    result = (num1 > num2) || (num1 > num3);
    printf("\n%d",result);
}
